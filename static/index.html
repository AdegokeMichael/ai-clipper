Platform Credentials
Add to .env and restart.
# Anthropic (required)
ANTHROPIC_API_KEY=sk-ant-...

# YouTube Shorts
YOUTUBE_CLIENT_SECRETS_FILE=./client_secret.json
YOUTUBE_TOKEN_FILE=./youtube_token.json

# Instagram Reels
INSTAGRAM_ACCESS_TOKEN=...
INSTAGRAM_ACCOUNT_ID=...
INSTAGRAM_PUBLIC_BASE_URL=http://your-server:8000

# Facebook Reels
FACEBOOK_ACCESS_TOKEN=...
FACEBOOK_PAGE_ID=...

# TikTok (pending API approval)
TIKTOK_ACCESS_TOKEN=...
TIKTOK_CLIENT_KEY=...
TIKTOK_CLIENT_SECRET=...

# Upload-Post (TikTok without API approval)
UPLOADPOST_API_KEY=your_api_key
UPLOADPOST_PROFILE=your_profile_name
# Verify: GET /uploadpost/verify
